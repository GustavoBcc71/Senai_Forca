#include <stdio.h>

// Variável global
int pontuacao = 100;

// Função para mostrar a barra de progresso
void mostrarBarraDeProgresso(int pontuacao) {
    int total_barras = 10;
    int preenchidas = pontuacao * total_barras / 100;
    if (preenchidas > total_barras) preenchidas = total_barras;
    if (preenchidas < 0) preenchidas = 0;

    printf("Progresso: ");
    for (int i = 0; i < preenchidas; i++) printf("#");
    for (int i = 0; i < total_barras - preenchidas; i++) printf("-");
    printf(" %d%%\n", pontuacao);
}

// Função para avaliar desempenho
const char* avaliarDesempenho(int pontuacao) {
    if (pontuacao >= 80) return "Excelente";
    else if (pontuacao >= 50) return "Bom";
    else return "Regular";
}

int main() {
    int tentativas = 7;
    int acertos = 0, erros = 0;
    int resposta;

    printf("Digite 1 para acerto ou 0 para erro em cada tentativa.\n");

    for (int i = 0; i < tentativas; i++) {
        printf("Tentativa %d/%d - Acertou? (1 = Sim, 0 = Não): ", i + 1, tentativas);
        scanf("%d", &resposta);

        if (resposta == 1) {
            pontuacao += 10;
            acertos++;
        } else if (resposta == 0) {
            pontuacao -= 2;
            erros++;
        } else {
            printf("Entrada inválida! Tente novamente.\n");
            i--; // repetir a tentativa
            continue;
        }

        mostrarBarraDeProgresso(pontuacao);
    }

    // Resultado final
    printf("\nAcertos: %d/%d → Pontuação: %d\n", acertos, tentativas, pontuacao);
    printf("Desempenho: %s\n", avaliarDesempenho(pontuacao));

    return 0;
}
